create table if not exists public.hvac_leads (

    created_at timestamptz default timezone('utc'::text, now()) not null,
    updated_at timestamptz default timezone('utc'::text, now()) not null,
    owner_id uuid references auth.users(id) on delete cascade, -- Nullable to support unassigned inbound leads safely
    customer_name text not null,
    phone text not null constraint phone_length_check check (char_length(phone) >= 10 and char_length(phone) <= 20),
    email text,
    city text not null,
    service_type text not null,
    property_type text not null,
    issue_description text not null constraint issue_length_check check (char_length(issue_description) >= 20 and char_length(issue_description) <= 2000),
    lead_quality text not null constraint lead_quality_check check (lead_quality in ('LOW', 'MEDIUM', 'HIGH')),
    urgency text not null constraint urgency_check check (urgency in ('LOW', 'NORMAL', 'HIGH', 'EMERGENCY')),
    estimated_job_value text not null constraint job_value_check check (estimated_job_value in ('LOW', 'MEDIUM', 'HIGH')),
    customer_intent text not null constraint intent_check check (customer_intent in ('UNKNOWN', 'SHOPPING', 'READY_TO_BUY')),
    recommended_response_time text not null,
    service_category text not null,
    summary text not null,
    recommended_action text not null,
    lead_score integer not null constraint lead_score_check check (lead_score >= 0 and lead_score <= 100),
    priority text not null constraint priority_check check (priority in ('LOW', 'MEDIUM', 'HIGH', 'CRITICAL')),
    status text default 'NEW'::text not null constraint status_check check (status in ('NEW', 'CONTACTED', 'SCHEDULED', 'COMPLETED', 'LOST')),
    source text default 'Website'::text not null constraint source_check check (source in ('Website', 'Google Ads', 'Facebook', 'Referral', 'Phone', 'Manual'))
);

-- 5. Create public.lead_events Table (Audit Timeline)
create table if not exists public.lead_events (

    lead_id uuid references public.hvac_leads(id) on delete cascade not null,
    event_type text not null constraint event_type_check check (

    ),
    description text not null,
    metadata jsonb default '{}'::jsonb,
    created_by uuid references auth.users(id) on delete set null,
    created_at timestamptz default timezone('utc'::text, now()) not null
);

-- 6. Create public.lead_notes Table (Dispatcher Notes)

    created_at timestamptz default timezone('utc'::text, now()) not null

-- 7. Create public.appointments Table (Site Visit Scheduling)
create table if not exists public.appointments (
    id uuid default gen_random_uuid() not null primary key,
    lead_id uuid references public.hvac_leads(id) on delete cascade not null,
    appointment_date date not null,
    appointment_time time not null,
    appointment_type text not null,
    status text not null constraint appointment_status_check check (status in ('Scheduled', 'Confirmed', 'Rescheduled', 'Completed', 'Cancelled')),
    notes text,
    created_at timestamptz default timezone('utc'::text, now()) not null,
    updated_at timestamptz default timezone('utc'::text, now()) not null
);

-- 8. Create public.reminders Table (Dispatcher Follow-ups)
create table if not exists public.reminders (
    id uuid default gen_random_uuid() not null primary key,
    lead_id uuid references public.hvac_leads(id) on delete cascade not null,
    reminder_date date not null,
    reminder_time time not null,
    priority text not null constraint reminder_priority_check check (priority in ('LOW', 'MEDIUM', 'HIGH', 'CRITICAL')),
    message text not null,
    status text default 'Pending'::text not null constraint reminder_status_check check (status in ('Pending', 'Completed', 'Overdue')),
    created_at timestamptz default timezone('utc'::text, now()) not null,
    updated_at timestamptz default timezone('utc'::text, now()) not null
);

-- 9. Create Highly Performance-Optimized Database Indexes
create index if not exists idx_leads_owner on public.hvac_leads(owner_id);
create index if not exists idx_leads_priority on public.hvac_leads(priority);
create index if not exists idx_leads_status on public.hvac_leads(status);

-- 10. Database Trigger Automation Functions
create or replace function public.handle_updated_at()
returns trigger as $$
begin
    new.updated_at = timezone('utc'::text, now());
    return new;
end;
$$ language plpgsql;

create or replace function public.log_lead_creation_event()
returns trigger as $$
begin
    -- Auto log LEAD_CREATED
    insert into public.lead_events (lead_id, event_type, description, metadata)
    values (
        new.id,
        'LEAD_CREATED',
        'New lead ingested from ' || new.source || ' for customer ' || new.customer_name,
        jsonb_build_object('customer_name', new.customer_name, 'source', new.source, 'service_type', new.service_type)
    );
    
    -- Auto log AI_ANALYZED if summary is parsed
    if new.summary is not null and new.summary <> '' then
        insert into public.lead_events (lead_id, event_type, description, metadata)
        values (
            new.id,
            'AI_ANALYZED',
            'AI successfully qualified the lead for ' || new.customer_name || ' (Score: ' || new.lead_score || ')',
            jsonb_build_object('lead_score', new.lead_score, 'priority', new.priority, 'quality', new.lead_quality)
        );
    end if;
    
    return new;
end;
$$ language plpgsql;

# HVAC AI Operations Platform - Database Deployment Guide
**Version: 1.0.0**
**Database: PostgreSQL (Supabase)**

This guide walks you through setting up a fresh Supabase database instance and seeding realistic demo data for the HVAC AI Operations Platform.

---

## 1. Create a New Supabase Project
1. Log in to the [Supabase Dashboard](https://supabase.com).
2. Click on **New Project** and select your organization.
3. Configure the project:
   - **Name:** `hvac-ai-operations`
   - **Database Password:** *Use a secure password containing capitals, numbers, and special characters.*
   - **Region:** Select the region closest to your operational area.
4. Click **Create new project** and wait 2 minutes for provisioning to complete.

---

## 2. Execute SQL Schema Installer
1. In your Supabase sidebar, navigate to the **SQL Editor** (represented by the SQL terminal icon).
2. Click on **New Query** to open an empty SQL terminal.
3. Copy the entire contents of the `/database/install.sql` file in this project.
4. Paste the SQL code into the terminal and click **Run** (or press `Cmd + Enter` / `Ctrl + Enter`).
5. Ensure the terminal output displays `Success` with no errors.

---

## 3. Seed Realistic HVAC Demo Data
1. Open a **New Query** window in the Supabase SQL Editor.
2. Copy the entire contents of the `/database/seed.sql` file.
3. Paste the SQL code into the terminal and click **Run**.
4. This will populate the console with 5 highly qualified leads, scheduled visits, and follow-ups.

---

## 4. Verify Database Integrity
1. Open a **New Query** window in the SQL Editor.
2. Copy the contents of `/database/verify.sql`.
3. Paste and click **Run**.
4. Review the results:
   - Ensure `rls_enabled` is listed as `true` for all tables.
   - Confirm that the total ingested leads count is registered as `5`.

---

## 5. Safe Reset (Optional)
If you need to wipe out the demo data to prepare the console for a clean production client but keep the database schemas intact:
1. Open a **New Query** window in the SQL Editor.
2. Paste and run `/database/reset.sql`.

---

## 6. Configure Local Environment
Copy your Supabase project API keys from **Project Settings** -> **API** and paste them into your local `.env.local` file:

NEXT_PUBLIC_SUPABASE_URL=your_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_anon_key
